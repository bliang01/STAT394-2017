	\begin{eqnarray}\ok
		\exp(x)&=& \sum_{n = 0}^\infty \frac{x^n}{n!}
	\\\ok
		log(1 - x)&=& -\sum_{n = 1}^\infty \frac{x^n}{n}
	\\\ok
		log(1 + x) &=& \sum_{n = 1}^\infty (-1)^{n+1} \frac{x^n}{n}
	\\\ok
		\frac{1}{1 - x} &=& \sum_{n = 0}^\infty x^n
	\end{eqnarray}
}
